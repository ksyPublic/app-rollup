#options

OPTION TYPE DEFAULT DATA ATTRIBUTE DESCRIPTION
openClass string "active" data-select-open-class 셀렉트 오픈 시 붙일 클래스 명
animate boolean false data-select-animate 애니메이션 여부
directionChange function null 셀렉트 오픈 시 오픈되어야 할 방향을 처리하는 hook 함수 입니다.

인스턴스 획득

								
								var element = document.querySelector("@selector");
								var Select = HitaiUI.Select.get(element);
								// 초기화되어 있지 않다면 null을 반환합니다.
								
							
								
								var Select = HitaiUI.Select.get($("@selector"));

🟢 @selector data-select="select" 이 선언되어있는 엘리먼트 입니다.

#selector

data-select="select"

속성이 선언된 엘리먼트는 셀렉트박스 컴포넌트로 초기화됩니다.

#methods

.init()

셀렉트박스를 초기화 합니다.

.disable(isDisable)

셀렉트박스를 활성화/비활성화 처리합니다.

.value()

선택된 아이템의의 value를 반환합니다.

									
										var Select = HitaiUI.Select.get(document.querySelector('#Select'));
										Select.value();
									
								
									
										HitaiUI.Select.get($("#Select")).value();

.text()

선택된 아이템의 textContent를 반환합니다.

.index()

선택된 아이템의 index를 반환합니다.

.selectValue(value)

value값을 통해 아이템을 선택합니다.

									
										var element = document.querySelector("#Select");
										HitaiUI.Select.get(element).selectValue(1);
									
								
									
										HitaiUI.Select.get($("#Select")).selectValue(1);

.selectIndex(index)

index값을 통해 아이템을 선택합니다.

									
										var element = document.querySelector("#Select");
										HitaiUI.Select.get(element).selectIndex(0); // 0번째 선택
									
								
									
										HitaiUI.Select.get($("#Select")).selectIndex(0); // 0번째 선택

.setDefault()

셀렉트 박스를 초기화 상태로 만듭니다(선택 되지 않은 상태로).

static

HitaiUI.Select.get(element, initialize = false)

셀렉트박스의 인스턴스를 반환합니다.

initialize 값을 true로 호출하면 초기화되지 않은 대상을 초기화 후 인스턴스를 반환합니다.

static

HitaiUI.Select.getInstances()

현재 화면에 존재하는 셀렉트박스의 모든 인스턴스를 반환합니다.

								
								var instanceList = HitaiUI.Select.getInstances();
								// output [] or [Select, Select, Select]

#events

HitaiUI.Select.EVENT.CHANGE

셀렉트박스의 선택값이 변경될 때 트리거 됩니다.

                    
                      var element = document.querySelector("#select-box");
                      var Select = HitaiUI.Select.get(element);
                      element.addEventListener(HitaiUI.Select.EVENT.CHANGE, function() {
                        Select.value();    // selected value
                        Select.text();     // selected text
                        Select.index();    // selected index
                      });
                    
								
									
										$("#select-box").on(HitaiUI.Select.EVENT.CHANGE, function() {
											HitaiUI.Select.get($(this)).value();   // selected value
											HitaiUI.Select.get($(this)).text();    // selected text
											HitaiUI.Select.get($(this)).index();   // selected index
										})

#examples

활성화, 비활성화 처리

필드명

								
									var Select = HitaiUI.Select.get(("#disableExample"));
									$("#btnDisabled").on("click", function () {
										Select.disable(true);
									});

									$("#btnEnabled").on("click", function () {
										Select.disable(false);
									});

상태가져오기 ( 데이터 받아오기 )

필드명
								
                  var selectElement = $("#example1");
                  var Select = HitaiUI.Select.get(selectElement);

                  // Select change event
                  selectElement.on(HitaiUI.Select.EVENT.CHANGE, function (e) {
                    var value = Select.value();
                    var index = Select.index();
                    var text = Select.text();
                    alert(value + " ,  " + index + " , " + text);
                  })

상태변경하기 ( 데이터 셋팅하기 )

필드명

data-select="value" 인풋에 직접 value값을 넣으면 셀렉트박스 컴포넌트가 자동으로 업데이트 합니다.

옵션 아이템에 태그가 포함되어있으면 selectText 메서드를 사용할 수 없습니다.

옵션 아이템에 태그가 포함되어있으면 data-select="value" 히든 인풋의 text 값으로 업데이트할 수 없습니다.

								
									var Select = HitaiUI.Select.get("#example2");

									// .selectValue("value3")
									$("#btnSetValue").on("click", function () {
										Select.selectValue("value3");
									});

									// .selectIndex(1)
									$("#btnSetIndex").on("click", function () {
										Select.selectIndex(1);
									});

									//.selectText("선택 항목5")
									$("#btnSetIndex2").on("click", function () {
										HitaiUI.Select.get(("#example2")).selectText("선택 항목5");
									});

									// 히든인풋에 데이터 직접 셋팅 - value 값
									$("#btnSetInput").on("click", function () {
										$('#example2 [data-select="value"]').attr('value', "value4")
									});

									// 히든인풋에 데이터 직접 셋팅 - text 값
									$("#btnSetInput").on("click", function () {
										$('#example2 [data-select="value"]').attr('text', "선택 항목1")
									});

									// 셀렉트박스 비활성화
									$("#btnDisabled").on("click", function () {
										HitaiUI.Select.get(("#example2")).disable(true);
									});

									// 셀렉트박스 활성화
									$("#btnEnabled").on("click", function () {
										HitaiUI.Select.get(("#example2")).disable(false);
									});
