Dialog

다이얼로그 컨텐츠를 구성합니다.

#options

OPTION TYPE DEFAULT DESCRIPTION
openClass string "open" 다이얼로그 오픈 시 붙일 클래스 명
closeClass string "close" 다이얼로그 닫힘 시 붙일 클래스 명
animate boolean true 애니메이션 여부
destroy boolean false 다이얼로그 닫힘 시 삭제 여부

인스턴스 획득

- 다이얼로그는 직접 초기화할 수 있지만 대게의 경우 트리거 data-dialog-trigger 를 통해서 클릭 시점에 자동으로 초기화합니다.

								
								var element = document.querySelector("@selector");
								var dialog = HitaiUI.Dialog.get(element);
								// 다이얼로그가 초기화되어 있지 않다면 null을 반환합니다.
								
							
								
								var dialog = HitaiUI.Dialog.get($("@selector"), true);

🟢@selector role="dialog" 이 선언되어있는 엘리먼트 입니다.

#selector

data-dialog-trigger="@target"

속성이 선언된 엘리먼트 클릭 시 @target 다이얼로그를 오픈합니다.

data-dialog-close

속성이 선언된 엘리먼트에 닫기 이벤트를 자동으로 바인딩합니다.

#methods

.open()

다이얼로그를 오픈합니다.

.close()

다이얼로그를 닫습니다.

.getElement()

다이얼로그의 엘리먼트를 반환합니다.

                  
                    var dialog1 = $('#uiDialog1');
                    HitaiUI.Dialog.get(dialog1, true).getElement();
                    // 해당 다이얼로그 엘리먼트 반환됨

.updateOption()

다이얼로그의 옵션을 업데이트합니다.

                  
                    var dialog1 = $('#uiDialog1');
                    var dialogInstance = HitaiUI.Dialog.get(dialog, true);
                    dialogInstance.updateOption({ animate: false });

.destroy()

다이얼로그를 삭제합니다.

이 메서드가 호출되면 엘리먼트까지 모두 삭제됩니다.

static

HitaiUI.Dialog.trigger(event)

document 감시 트리거

static

HitaiUI.Dialog.closeAll()

화면에 열려있는 모든 다이얼로그를 닫습니다.

static

HitaiUI.Dialog.get(element, initialize = false)

다이얼로그의 인스턴스를 반환합니다.

initialize 값을 true로 호출하면 초기화되지 않은 대상을 초기화 후 인스턴스를 반환합니다.

static

HitaiUI.Dialog.getInstances()

현재 화면에 존재하는 다이얼로그의 모든 인스턴스를 반환합니다.

                  
                  var instanceList = HitaiUI.Dialog.getInstances();
                  // output [] or [Dialog, Dialog, Dialog]

#events

HitaiUI.Dialog.EVENT.INIT

다이얼로그가 초기화되면 트리거 됩니다.

									
										// event type string: "ui.dialog.init"
										var element = document.querySelector('#dialog');
										element.addEventListener(HitaiUI.Dialog.EVENT.INIT, function(event) {
										var component = event.component; //component instance
										});
									
								
									
										// event type string: "ui.dialog.init"
										var $element = $('#dialog');
										$element.on(HitaiUI.Dialog.EVENT.INIT, function(event) {
										var component = event.component; //component instance
										});

HitaiUI.Dialog.EVENT.OPEN

다이얼로그가 열릴 때 트리거 됩니다.

									
										// event type string: "ui.dialog.open"
										var element = document.querySelector('#dialog');
										element.addEventListener(HitaiUI.Dialog.EVENT.OPEN, function(event) {console.log('열릴 때 발생할 처리')});
									
								
									
										// event type string: "ui.dialog.open"
										var $element = $('#dialog');
										$element.on(HitaiUI.Dialog.EVENT.OPEN, function(event) {console.log('열릴 때 발생할 처리')});

HitaiUI.Dialog.EVENT.OPENED

다이얼로그가 열린 후 트리거 됩니다.(애니메이션 종료 후 트리거)

HitaiUI.Dialog.EVENT.CLOSE

다이얼로그가 닫힐 때 트리거 됩니다.

HitaiUI.Dialog.EVENT.CLOSED

다이얼로그가 닫힌 후 트리거 됩니다.(애니메이션 종료 후 트리거).

기본적인 다이얼로그 사용방법입니다.

팝업제목

컨텐츠
								
								// 자동으로 초기화되기 때문에 따로 처리해야 할 코드 없습니다.

버튼 클릭 없이 직접 초기화하여 바로 오픈하는 방법

data-dialog-trigger 버튼을 사용하지 않고 다이얼로그를 오픈합니다.

팝업제목

컨텐츠
                    
                      //----------------------------------------------------------------------
                      // Open button
                      HitaiUI.Dialog.get(document.querySelector("#uiDialog2"), true).open();
                      //----------------------------------------------------------------------
  
  
                      //----------------------------------------------------------------------
                      // Open(이벤트 사용) button
                      var el = document.querySelector("#uiDialog2");
                      var dialog = HitaiUI.Dialog.get(el, true); // 다이얼로그 생성
                      // 닫힘 이벤트 등록 후 사용
                      el.addEventListener(HitaiUI.Dialog.EVENT.CLOSED, function closed() {
                        // 등록된 이벤트는 반드시 삭제
                        el.removeEventListener(HitaiUI.Dialog.EVENT.CLOSED, closed);
                          HitaiUI.Alert(HitaiUI.Dialog.EVENT.CLOSED + " >> 이벤트 실행!");
                        });
                      dialog.open();
                      //----------------------------------------------------------------------
  
  
                      //----------------------------------------------------------------------
                      HitaiUI.Dialog.get(HitaiUI.Event.one(document.querySelector("#uiDialog2"), HitaiUI.Dialog.EVENT.CLOSED, function() {
                        HitaiUI.Alert(HitaiUI.Dialog.EVENT.CLOSED + " >> 이벤트 실행!");
                      }), true).open();
                      //----------------------------------------------------------------------
  
                    
                  
                    
                      //----------------------------------------------------------------------
                      // Open button
                      HitaiUI.Dialog.get($("#popExample3"), true).open();
                      //----------------------------------------------------------------------
  
  
                      //----------------------------------------------------------------------
                      // Open(이벤트 사용) button
                      var $el = $("#popExample3");
                      var dialog = HitaiUI.Dialog.get($el, true); // 다이얼로그 생성
                      $el.one(HitaiUI.Dialog.EVENT.CLOSED, function() {
                        HitaiUI.Alert(HitaiUI.Dialog.EVENT.CLOSED + " >> 이벤트 실행!");
                      });
                      dialog.open();
                      //----------------------------------------------------------------------
  
  
                      //----------------------------------------------------------------------
                      HitaiUI.Dialog.get($("#popExample3").one(HitaiUI.Dialog.EVENT.CLOSED, function() {
                        HitaiUI.Alert(HitaiUI.Dialog.EVENT.CLOSED + " >> 이벤트 실행!");
                      }), true).open();
                      //----------------------------------------------------------------------
